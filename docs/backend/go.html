零值

  • 布尔类型是 false, 数字类型是 0, 字符串类型是 ""
  • 指针、函数、接口、slice、map、chan 的零值是 nil
package main
 
import "fmt"
 
func main() {
	var a []int
	fmt.Printf("len(a):%d, cap(a):%d, a==nil:%v\n", len(a), cap(a), a == nil) // 0 0 true
 
	// 遍历 nil 切片, 不会panic
	for _, v := range a {
		fmt.Println(v)
	}
	// a[0:0] // 不会 panic
	// a[0:1] // 会 panic
 
	var b map[string]string
	if val, ok := b["key"]; ok { // 不会 panic
		fmt.Println(val)
	}
 
	for k, v := range b { // 不会 panic
		fmt.Println(k, v)
	}
	delete(b, "foo")                                       // 不会 panic
	fmt.Printf("len(b):%d, b==nil:%v\n", len(b), b == nil) // 0 true
}

slice, map, chan

  • slice 底层是结构体
  • map 和 chan 底层是指针
  • go 函数传参, 都是值拷贝
type slice struct {
  array unsafe.Pointer
  len   int
  cap   int
}
 
// Bad case
func push(s []int){
  s = append(s, 1)
}

深拷贝

深拷贝 slice

import "slices"
 
s1 := []int{1, 2, 3}
s2 := append([]int{}, s1...)
 
s1 := []int{1, 2, 3}
s2 := make([]int, 0, len(s1))
copy(s2, s1)
 
s1 := []int{1, 2, 3}
s2 := slices.Clone(s1)

深拷贝 map

import "maps"
 
m1 := map[string]int{"a": 1, "b": 2}
m2 := maps.Clone(m1)
